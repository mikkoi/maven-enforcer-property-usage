31          log = logger;
32          charset = cset;

35      /**
36       * @param filenames Collection of file names to read properties from.
37       * @return Map of definitions and how many times they are defined.
38       */
39      @Nonnull
40      Map<String, Integer> readPropertiesFromFilesWithoutCount(@Nonnull final Collection<String> filenames)
41              throws IOException {
42          Map<String, Integer> results = new HashMap<>();
43          for (String filename : filenames) {
44              File file = new File(filename);
45              Properties properties = new Properties();
46              log.debug("PropertyFiles:readPropertiesFromFilesWithoutCount() Reading file " + filename + ".");
47              try (InputStream inputStream = new FileInputStream(file)) {
48                  properties.load(inputStream);
49                  for (String name : properties.stringPropertyNames()) {
50                      log.debug("    Reading property " + name + ".");
51                      results.put(name, 1);
52                  }
53              }
54          }
55          return results;
56      }

58      /**
59       * Read properties with our own reading routine and count
60       * how many times they are used.
61       *
62       * @param filenames Collection of file names to read properties from.
63       * @return Map of definitions and how many times they are defined.
64       */
65      @SuppressWarnings({
66              "squid:S3776", // Cognitive Complexity of methods should not be too high
67              "squid:S134",  // Control flow statements "if", "for", "while", "switch" and "try" should not be nested too deeply
68              "squid:S135"   // Loops should not contain more than a single "break" or "continue" statement
69      })
70      @Nonnull
71      Map<String, Integer> readPropertiesFromFilesWithCount(@Nonnull final Collection<String> filenames)
72              throws IOException {
73          Map<String, Integer> results = new HashMap<>();
74          Pattern commentLineP = Pattern.compile("^[\\s]{0,}[\\#\\!]{1}.{0,}$");
75          log.debug("commentLineP:" + commentLineP.pattern());
76          Pattern simplePropertyLineP = Pattern.compile("^[\\s]{0,}([^=:]{1,})[=:]{1}.{0,}$", Pattern.COMMENTS | Pattern.UNICODE_CHARACTER_CLASS);
77          log.debug("simplePropertyLineP:" + simplePropertyLineP.pattern());
78          Pattern notSimplePropertyLineP = Pattern.compile("^[\\s]{0,}([\\S]{1,})[\\s]{0,}.{0,}$", Pattern.COMMENTS | Pattern.UNICODE_CHARACTER_CLASS);
79          log.debug("simplePropertyLineP:" + simplePropertyLineP.pattern());
80          Pattern multiLineP = Pattern.compile("\\\\{1}[\\s]{0,}$");
81          log.debug("multiLineP:" + multiLineP);
82          for (String filename : filenames) {
83              log.debug("Reading property file '" + filename + "'.");
84              List<String> rows = Files.readAllLines(Paths.get(filename), charset);
85              boolean readingMultiLineDefinition = false;
86              for ( String row : rows) {
87                  log.debug("    Reading property row '" + row + "'.");
88                  Matcher commentLineM = commentLineP.matcher(row);
89                  if (commentLineM.find()) {
90                      log.debug("        This is comment line.");
91                      continue;
92                  }
93                  Matcher multiLineM = multiLineP.matcher(row);
94                  if (multiLineM.find()) {
95                      if( readingMultiLineDefinition) {
96                          log.debug("        This is multirow (not first row)");
97                          continue;
98                      } else {
99                          log.debug("        This is multirow (first row).");
100                         readingMultiLineDefinition = true;
101                     }
102                 } else {
103                     if( readingMultiLineDefinition) {
104                         log.debug("        This is multirow (last row).");
105                         readingMultiLineDefinition = false;
106                         continue;
107                     }
108                 }
109                 Matcher simplePropertyLineM = simplePropertyLineP.matcher(row);
110                 if (simplePropertyLineM.find()) {
111                     log.debug("        This is simple property line.");
112                     final String name = simplePropertyLineM.group(1).trim();
113                     storePropertyName(name, results);
114                     continue;
115                 }
116                 Matcher notSimplePropertyLineM = notSimplePropertyLineP.matcher(row);
117                 if(notSimplePropertyLineM.find()) {
118                     log.debug("        This is not simple property line.");
119                     final String name = notSimplePropertyLineM.group(1).trim();
120                     storePropertyName(name, results);
121                     continue;
122                 }
123                 log.debug("        This row matched nothing,  propably empty or multiline continuation.");
124             }
125         }
126         return results;
127     }
128 
129     private void storePropertyName(@Nonnull final String name, @Nonnull Map<String,Integer> properties) {
130         log.debug("            Reading property " + name + ".");
131         if (!properties.containsKey(name)) {
132             log.debug("            Not defined before.");
133             properties.put(name, 1);
134         } else {
135             log.debug("            Defined before. Update counter.");
136             properties.replace(name, properties.get(name) + 1);
137         }
138     }
